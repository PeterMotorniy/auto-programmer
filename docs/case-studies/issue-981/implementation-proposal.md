# Implementation Proposal: `--keep-tool-context-in-repository` Option

## Overview

This document outlines the technical implementation plan for the `--keep-tool-context-in-repository` option in hive-mind's solve.mjs tool.

## Feature Specification

### Command Line Interface

```bash
# Basic usage - enable context persistence
./solve.mjs "https://github.com/org/repo/issues/123" --keep-tool-context-in-repository

# With alias (shorter form)
./solve.mjs "https://github.com/org/repo/issues/123" --keep-context

# Combined with other options
./solve.mjs "https://github.com/org/repo/issues/123" --tool claude --keep-context --attach-logs
```

### Configuration

Add to `solve.config.lib.mjs`:

```javascript
.option('keep-tool-context-in-repository', {
  alias: 'keep-context',
  type: 'boolean',
  describe: 'Synchronize AI session context to repository .knowledge folder',
  default: false
})
.option('context-summary-model', {
  type: 'string',
  describe: 'Model to use for session summarization',
  default: 'haiku'
})
.option('knowledge-folder', {
  type: 'string',
  describe: 'Folder path for storing context (relative to repo root)',
  default: '.knowledge'
})
```


### Implementation Files

#### 1. New Library: `src/context-persistence.lib.mjs`

```javascript
/**
 * Context persistence library for storing AI session context in repositories.
 */

import path from 'path';
import fs from 'fs/promises';

/**
 * Get source session file path for a tool
 */
export function getSessionSourcePath({ tool, sessionId, tempDir }) {
  const os = require('os');
  const homeDir = os.homedir();

  switch (tool) {
    case 'claude':
      const projectDirName = tempDir.replace(/\//g, '-');
      return path.join(homeDir, '.claude', 'projects', projectDirName, `${sessionId}.jsonl`);
    case 'codex':
      return path.join(homeDir, '.codex', 'sessions', `${sessionId}.jsonl`);
    case 'opencode':
      return path.join(homeDir, '.opencode', 'conversations', `${sessionId}.jsonl`);
    case 'agent':
      // Agent sessions may be stored differently
      return null;
    default:
      return null;
  }
}

/**
 * Initialize the .knowledge folder structure
 */
export async function initializeKnowledgeFolder(repoPath, folderName = '.knowledge') {
  const knowledgePath = path.join(repoPath, folderName);

  // Create directories
  await fs.mkdir(path.join(knowledgePath, 'raw', '.claude', 'sessions'), { recursive: true });
  await fs.mkdir(path.join(knowledgePath, 'raw', '.codex', 'sessions'), { recursive: true });
  await fs.mkdir(path.join(knowledgePath, 'raw', '.opencode', 'conversations'), { recursive: true });
  await fs.mkdir(path.join(knowledgePath, 'summaries'), { recursive: true });

  // Create .gitignore for raw folder
  const gitignorePath = path.join(knowledgePath, '.gitignore');
  const gitignoreContent = `# Raw session files - kept locally only
raw/

# Session JSONL files may contain sensitive information
*.jsonl
`;
  await fs.writeFile(gitignorePath, gitignoreContent);

  // Create README.md
  const readmePath = path.join(knowledgePath, 'README.md');
  const readmeContent = `# Project Knowledge Base

This folder contains accumulated AI session context and learnings.

## Structure

- \`summaries/\` - AI-generated summaries of sessions (committed to git)
- \`patterns.md\` - Recurring patterns identified across sessions
- \`decisions.md\` - Key architectural decisions with rationale
- \`raw/\` - Full session files (local only, gitignored)

## Usage

Reference these files in your CLAUDE.md:

\`\`\`markdown
See @.knowledge/patterns.md for established patterns.
See @.knowledge/decisions.md for architectural decisions.
\`\`\`

---
*Generated by hive-mind with \`--keep-tool-context-in-repository\`*
`;
  await fs.writeFile(readmePath, readmeContent);

  // Create empty patterns and decisions files
  await fs.writeFile(path.join(knowledgePath, 'patterns.md'), '# Accumulated Patterns\n\n*No patterns recorded yet.*\n');
  await fs.writeFile(path.join(knowledgePath, 'decisions.md'), '# Architectural Decisions\n\n*No decisions recorded yet.*\n');

  return knowledgePath;
}

/**
 * Copy session file to knowledge folder
 */
export async function copySessionToKnowledge({ tool, sessionId, tempDir, repoPath, knowledgeFolder = '.knowledge' }) {
  const sourcePath = getSessionSourcePath({ tool, sessionId, tempDir });
  if (!sourcePath) {
    return { success: false, reason: 'Unknown tool or session path' };
  }

  try {
    await fs.access(sourcePath);
  } catch {
    return { success: false, reason: 'Session file not found' };
  }

  const toolFolderMap = {
    claude: path.join('.claude', 'sessions'),
    codex: path.join('.codex', 'sessions'),
    opencode: path.join('.opencode', 'conversations')
  };

  const destDir = path.join(repoPath, knowledgeFolder, 'raw', toolFolderMap[tool]);
  await fs.mkdir(destDir, { recursive: true });

  const destPath = path.join(destDir, `${sessionId}.jsonl`);
  await fs.copyFile(sourcePath, destPath);

  return { success: true, sourcePath, destPath };
}

/**
 * Generate session summary using AI
 */
export async function generateSessionSummary({ sessionId, sessionContent, issueUrl, model = 'haiku' }) {
  // This would use the same AI infrastructure as solve.mjs
  // For now, return a template that can be filled in manually or by AI

  const summary = {
    sessionId,
    issueUrl,
    date: new Date().toISOString().split('T')[0],
    keyDecisions: [],
    patternsIdentified: [],
    lessonsLearned: [],
    filesModified: [],
    toolsUsed: []
  };

  return summary;
}

/**
 * Save session summary to markdown file
 */
export async function saveSummary({ repoPath, issueId, summary, knowledgeFolder = '.knowledge' }) {
  const summariesDir = path.join(repoPath, knowledgeFolder, 'summaries');
  await fs.mkdir(summariesDir, { recursive: true });

  const filename = `${summary.date}-issue-${issueId}.md`;
  const filePath = path.join(summariesDir, filename);

  const content = `# Session Summary: Issue #${issueId}

**Date**: ${summary.date}
**Session ID**: ${summary.sessionId}
**Issue**: ${summary.issueUrl}

## Key Decisions

${summary.keyDecisions.length > 0 ? summary.keyDecisions.map(d => `- ${d}`).join('\n') : '*No decisions recorded*'}

## Patterns Identified

${summary.patternsIdentified.length > 0 ? summary.patternsIdentified.map(p => `- ${p}`).join('\n') : '*No patterns identified*'}

## Lessons Learned

${summary.lessonsLearned.length > 0 ? summary.lessonsLearned.map(l => `- ${l}`).join('\n') : '*No lessons recorded*'}

## Files Modified

${summary.filesModified.length > 0 ? summary.filesModified.map(f => `- \`${f}\``).join('\n') : '*No files tracked*'}

---
*Generated by hive-mind*
`;

  await fs.writeFile(filePath, content);
  return filePath;
}
```

#### 2. Integration in `solve.mjs`

Add after session completion (around line 1000):

```javascript
// Context persistence - copy session to repository
if (argv.keepToolContextInRepository && result.sessionId) {
  const contextLib = await import('./context-persistence.lib.mjs');

  try {
    // Initialize knowledge folder if needed
    await contextLib.initializeKnowledgeFolder(tempDir, argv.knowledgeFolder);

    // Copy raw session file
    const copyResult = await contextLib.copySessionToKnowledge({
      tool: argv.tool,
      sessionId: result.sessionId,
      tempDir,
      repoPath: tempDir,
      knowledgeFolder: argv.knowledgeFolder
    });

    if (copyResult.success) {
      await log(`📂 Session context saved to ${argv.knowledgeFolder}/`);
    }

    // Generate and save summary (future enhancement)
    // const summary = await contextLib.generateSessionSummary({ ... });
    // await contextLib.saveSummary({ ... });

  } catch (error) {
    await log(`⚠️ Could not persist session context: ${error.message}`, { level: 'warning' });
  }
}
```

## Risks and Mitigations

| Risk | Probability | Impact | Mitigation |
|------|-------------|--------|------------|
| Large session files | High | Medium | Rotation policy, cleanup scripts |
| Sensitive data exposure | Medium | High | Gitignore raw files, review summaries |
| Breaking changes | Low | Medium | Feature flag, gradual rollout |
| Performance impact | Low | Low | Async file operations |
